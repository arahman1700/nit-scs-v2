   - Extend `buildExtendedClient` to add a `$allOperations` extension that wraps `$transaction` calls with timing:
     After the existing `query: { $allModels: { ... } }` block in `$extends`, do NOT modify the existing soft-delete extension. Instead, chain a SECOND `$extends` call after the first one. Add timing to the query layer using a new extension:
     ```typescript
     function buildExtendedClient(base: PrismaClient): PrismaClient {
       const withSoftDelete = base.$extends({ query: { $allModels: { /* existing soft-delete */ } } });
       const withMetrics = withSoftDelete.$extends({
         query: {
           $allOperations({ operation, model, args, query }) {
             const start = performance.now();
             const result = query(args);
             if (result instanceof Promise) {
               return result.finally(() => {
                 const durationSec = (performance.now() - start) / 1000;
                 if (durationSec > 0.1) {
                   prismaTransactionDuration.observe({ operation: `${model}.${operation}` }, durationSec);
                 }
               });
             }
             return result;
           },
         },
       });
       return withMetrics as any as PrismaClient;
     }
     ```
     NOTE: Only observe queries over 100ms to avoid flooding the histogram with trivial queries. This keeps the metric focused on operations worth monitoring (transactions, complex queries).
