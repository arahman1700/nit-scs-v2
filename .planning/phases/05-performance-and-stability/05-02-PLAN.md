<objective>
Fix the bin cards computed endpoint N+1 query problem and add query timeout (PERF-01/PERF-03), then add Redis caching for master data lists, approval chain lookups, and permission matrix (PERF-07).

Purpose: The bin cards endpoint currently issues 3 individual queries per inventory level inside a Promise.all(levels.map(...)), creating O(3N) database round-trips. For a warehouse with 500 items this means 1500 queries per request. Batching these into 3 total queries fixes the performance hang. Adding Redis cache-aside to frequently-read reference data (master data, approval chains) reduces database load on the most common read paths.

Output: Optimized bin card endpoint, Redis-cached master data and approval chain lookups, with cache invalidation on mutations.
</objective>

    ```typescript
    /**
     * Express middleware that caches JSON responses in Redis.
     * Only applies to GET requests. POST/PUT/DELETE invalidate the cache.
     */
    function redisCacheMiddleware(cachePrefix: string, ttl: number) {
      return async (req: Request, res: Response, next: NextFunction) => {
        // Only cache GET requests
        if (req.method !== 'GET') {
          // Mutation — invalidate cache for this resource
          res.on('finish', () => {
            if (res.statusCode >= 200 && res.statusCode < 300) {
              invalidateCachePattern(`${cachePrefix}:*`).catch(() => {});
            }
          });
          return next();
        }

        // Build cache key from URL + query params
        const cacheKey = `${cachePrefix}:${req.originalUrl}`;

        try {
          const result = await cached(cacheKey, ttl, async () => {
            // Capture the response by wrapping res.json
            return new Promise<unknown>((resolve) => {
              const originalJson = res.json.bind(res);
              res.json = (body: unknown) => {
                resolve(body);
                return originalJson(body);
              };
              next();
            });
          });

          // If the promise resolved from cache (not from next()), send it
          if (!res.headersSent) {
            res.json(result);
          }
        } catch {
          next();
        }
      };
    }
    ```

    Create a small Express middleware factory in cache.ts (or directly in master-data.routes.ts):
    ```typescript
    import { getRedis, isRedisAvailable } from '../../../config/redis.js';

    const CACHE_PREFIX = 'nit-scs:cache:';

    function masterDataCacheMiddleware(resourceName: string) {
      return async (req: Request, res: Response, next: NextFunction) => {
        // Only cache list GETs (not detail GETs with /:id)
        if (req.method !== 'GET' || req.params.id) {
          return next();
        }

        if (!isRedisAvailable()) return next();
        const redis = getRedis();
        if (!redis) return next();

        const cacheKey = `${CACHE_PREFIX}master-data:${resourceName}:${req.url}`;

        try {
          const hit = await redis.get(cacheKey);
          if (hit) {
            res.setHeader('X-Cache', 'HIT');
            res.json(JSON.parse(hit));
            return;
          }
        } catch { /* fall through */ }

        // Cache MISS — intercept res.json to capture and cache the response
        const originalJson = res.json.bind(res);
        res.json = function(body: unknown) {
          // Cache the response asynchronously
          if (redis && res.statusCode >= 200 && res.statusCode < 300) {
            redis.setex(cacheKey, CacheTTL.MASTER_DATA, JSON.stringify(body)).catch(() => {});
          }
          res.setHeader('X-Cache', 'MISS');
          return originalJson(body);
        } as typeof res.json;

        next();
      };
    }
    ```
